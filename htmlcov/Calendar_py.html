35 """

37 You do not need to worry about what this function exactly does, nor create test cases for it. 

38 """ 

39 creds = None 

40 # The file token.pickle stores the user's access and refresh tokens, and is 

41 # created automatically when the authorization flow completes for the first 

42 # time. 

43 if os.path.exists('token.pickle'): 

44 with open('token.pickle', 'rb') as token: 

45 creds = pickle.load(token) 

46 

47 # If there are no (valid) credentials available, let the user log in. 

48 if not creds or not creds.valid: 

49 if creds and creds.expired and creds.refresh_token: 

50 creds.refresh(Request()) 

51 else: 

52 flow = InstalledAppFlow.from_client_secrets_file( 

53 'credentials.json', SCOPES) 

54 creds = flow.run_local_server(port=0) 

55 # Save the credentials for the next run 

56 with open('token.pickle', 'wb') as token: 

57 pickle.dump(creds, token) 

58 

59 return build('calendar', 'v3', credentials=creds)
